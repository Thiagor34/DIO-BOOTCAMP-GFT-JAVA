Alguns comando em GIT BASH

//verifica status
$ git status

//adiciona no controle de versão local
$ git add .

//realiza o commit para o arquivo
$ git commit -m "texto explicativo"

//mandar para a nuvem
$ git push origin main  

//faz um fetch das atualizações feitas em repositório online
$ git fetch origin

//faz um merge das atualizações feitas online para o trabalho local
$ git merge origin
